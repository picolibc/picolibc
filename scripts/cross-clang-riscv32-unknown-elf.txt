[binaries]
# Meson 0.53.2 doesn't use any cflags when doing basic compiler tests,
# so we have to add -nostdlib to the compiler configuration itself or
# early compiler tests will fail. This can be removed when picolibc
# requires at least version 0.54.2 of meson.
c = ['clang', '-target', 'riscv32-unknown-elf', '-nostdlib']
cpp = ['clang', '-target', 'riscv32-unknown-elf', '-nostdlib']
c_ld = 'eld'
cpp_ld = 'eld'
ar = 'llvm-ar'
as = 'clang'
nm = 'llvm-nm'
strip = 'llvm-strip'
readelf = 'llvm-readelf'
# only needed to run tests
exe_wrapper = ['sh', '-c', 'test -z "$PICOLIBC_TEST" || run-riscv "$@"', 'run-riscv']
[host_machine]
system = 'none'
cpu_family = 'riscv32'
cpu = 'riscv'
endian = 'little'
[properties]
# define libgcc to be empty, otherwise meson build configuration will add '-lgcc'
libgcc = '-lclang_rt.builtins'
skip_sanity_check = true
has_link_defsym = true
default_flash_addr = '0x80000000'
default_flash_size = '0x00400000'
default_ram_addr   = '0x80800000'
default_ram_size   = '0x00200000'
